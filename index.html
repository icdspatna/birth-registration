async function findAndNavigate() { const codeInput = document.getElementById('centerCode').value.trim().toUpperCase(); const status = document.getElementById('status'); if (!codeInput) { alert("Please enter a code"); return; } const webAppUrl = "https://script.google.com/macros/s/AKfycbzHEfpbMVsbfn59TS5jqMRO79KamfurTyBiB0s_cNGpNdG4z8HcuTwlO7iowpsqNr7r/exec"; status.innerText = "🔍 Searching Google Sheets..."; status.style.color = "blue"; try { const response = await fetch(webAppUrl); const csvData = await response.text(); const rows = csvData.split("\n"); let lat = "", lng = ""; for (let i = 0; i < rows.length; i++) { const columns = rows[i].split(",").map(c => c.trim().replace(/^"|"$/g, '')); // Check if Column A matches the Code if (columns[0] && columns[0].toUpperCase() === codeInput) { lat = columns[1]; lng = columns[2]; break; } } if (lat && lng) { status.innerText = "✅ Found! Opening Maps..."; status.style.color = "green"; // Correct Google Maps Directions URL const mapUrl = `https://google.com{lat},${lng}`; window.location.href = mapUrl; } else { status.innerText = `❌ Code '${codeInput}' not found in Sheet.`; status.style.color = "red"; } } catch (error) { console.error(error); status.innerText = "⚠️ Connection Error. Ensure Web App is 'Anyone'."; status.style.color = "orange"; } }
